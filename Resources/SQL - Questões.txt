Qual departamento tem o maior impacto no faturamento e porque?

SELECT nome, SUM("Valor") AS "Custos"
				
FROM rh_nota_fiscal AS nf

LEFT JOIN rh_departamento AS rd
ON rd.id = nf.departamento_id

GROUP BY nome

ORDER BY "Custos" DESC;

SELECT rd.nome, SUM("Pagamento") + (SUM("Hora_Extra") * 12) AS CustoSalario
				
FROM rh_colaborador AS colab

LEFT JOIN rh_folha_pag AS fpag
ON colab.folha_pag_id = fpag.id

LEFT JOIN rh_departamento AS rd
ON colab.departamento_id = rd.id

GROUP BY rd.nome;
